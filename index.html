Primer: Encoder – Decoder Architecture

Before training a language model, it helps to understand the two fundamental neural architectures that power modern NLP: encoders and decoders.The training script targets a tiny ~4M-parameter model that runs on CPU, but scales to the 1B-parameter target in the original Dante Project proposal. This primer explains both from the ground up, with runnable PyTorch code.

What Is a Neural Language Model?

A language model learns to predict the next token in a sequence. Given tokens x₁, …, xₜ, it estimates P(xₜ₊₁ | x₁, …, xₜ). The architecture determines how it processes that sequence: encoder-only, decoder-only, or the hybrid encoder-decoder topology of the original Transformer.

Encoder-Only Models (BERT, RoBERTa)

An encoder produces a contextual representation of every token using bidirectional self-attention.Best for: classification, NER, sentence embeddings. Cannot generate text on its own.

import torch, math
import torch.nn as nn

class SelfAttention(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super().__init__()
        assert embed_dim % num_heads == 0
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads
        self.q_proj = nn.Linear(embed_dim, embed_dim)
        self.k_proj = nn.Linear(embed_dim, embed_dim)
        self.v_proj = nn.Linear(embed_dim, embed_dim)
        self.out_proj = nn.Linear(embed_dim, embed_dim)

    def forward(self, x):
        B, T, C = x.shape
        q = self.q_proj(x).view(B, T, self.num_heads, self.head_dim).transpose(1, 2)
        k = self.k_proj(x).view(B, T, self.num_heads, self.head_dim).transpose(1, 2)
        v = self.v_proj(x).view(B, T, self.num_heads, self.head_dim).transpose(1, 2)
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.head_dim)
        attn = torch.softmax(scores, dim=-1)
        out = torch.matmul(attn, v).transpose(1, 2).contiguous().view(B, T, C)
        return self.out_proj(out)

class EncoderBlock(nn.Module):
    def __init__(self, embed_dim, num_heads, ff_dim, dropout=0.1):
        super().__init__()
        self.attn = SelfAttention(embed_dim, num_heads)
        self.ln1 = nn.LayerNorm(embed_dim)
        self.ffn = nn.Sequential(nn.Linear(embed_dim, ff_dim), nn.GELU(), nn.Linear(ff_dim, embed_dim))
        self.ln2 = nn.LayerNorm(embed_dim)
        self.dropout = nn.Dropout(dropout)

    def forward(self, x):
        x = x + self.dropout(self.attn(self.ln1(x)))
        x = x + self.dropout(self.ffn(self.ln2(x)))
        return x


class MiniEncoder(nn.Module):
    def __init__(self, vocab_size, embed_dim, num_heads, num_layers, ff_dim, max_len=512):
        super().__init__()
        self.tok_emb = nn.Embedding(vocab_size, embed_dim)
        self.pos_emb = nn.Embedding(max_len, embed_dim)
        self.blocks = nn.ModuleList([EncoderBlock(
            embed_dim, num_heads, ff_dim) for _ in range(num_layers)]
        )
        self.ln_f = nn.LayerNorm(embed_dim)

    def forward(self, input_ids):
        B, T = input_ids.shape
        pos = torch.arange(T, device=input_ids.device).unsqueeze(0)
        x = self.tok_emb(input_ids) + self.pos_emb(pos)
        for block in self.blocks:
            x = block(x)
        return self.ln_f(x)

Decoder-Only Models (GPT, LLaMA, Mistral)

A decoder generates text one token at a time using causal (masked) self-attention.The only structural change from the encoder is a single triu mask — that one line turns a representation model into a generative one.

class CausalSelfAttention(SelfAttention):
    def forward(self, x):
        B, T, C = x.shape
        q = self.q_proj(x).view(B, T, self.num_heads, self.head_dim).transpose(1, 2)
        k = self.k_proj(x).view(B, T, self.num_heads, self.head_dim).transpose(1, 2)
        v = self.v_proj(x).view(B, T, self.num_heads, self.head_dim).transpose(1, 2)
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.head_dim)
        mask = torch.triu(torch.ones(T, T, device=x.device), diagonal=1).bool()
        scores = scores.masked_fill(mask, float("-inf"))
        attn = torch.softmax(scores, dim=-1)
        out = torch.matmul(attn, v).transpose(1, 2).contiguous().view(B, T, C)
        return self.out_proj(out)


class DecoderBlock(nn.Module):
    def __init__(self, embed_dim, num_heads, ff_dim, dropout=0.1):
        super().__init__()
        self.attn = CausalSelfAttention(embed_dim, num_heads)
        self.ln1 = nn.LayerNorm(embed_dim)
        self.ffn = nn.Sequential(nn.Linear(embed_dim, ff_dim), nn.GELU(), nn.Linear(ff_dim, embed_dim))
        self.ln2 = nn.LayerNorm(embed_dim)
        self.dropout = nn.Dropout(dropout)

    def forward(self, x):
        x = x + self.dropout(self.attn(self.ln1(x)))
        x = x + self.dropout(self.ffn(self.ln2(x)))
        return x

class MiniDecoder(nn.Module):
    def __init__(self, vocab_size, embed_dim, num_heads, num_layers, ff_dim, max_len=2048):
        super().__init__()
        self.tok_emb = nn.Embedding(vocab_size, embed_dim)
        self.pos_emb = nn.Embedding(max_len, embed_dim)
        self.blocks = nn.ModuleList([DecoderBlock(embed_dim, num_heads, ff_dim) for _ in range(num_layers)])
        self.ln_f = nn.LayerNorm(embed_dim)
        self.lm_head = nn.Linear(embed_dim, vocab_size, bias=False)
        self.lm_head.weight = self.tok_emb.weight

    def forward(self, input_ids):
        B, T = input_ids.shape
        pos = torch.arange(T, device=input_ids.device).unsqueeze(0)
        x = self.tok_emb(input_ids) + self.pos_emb(pos)
        for block in self.blocks:
            x = block(x)
        return self.lm_head(self.ln_f(x))

    @torch.no_grad()
    def generate(self, input_ids, max_new_tokens=100, temperature=0.8, top_k=50):
        self.eval()
        for _ in range(max_new_tokens):
            logits = self.forward(input_ids)[:, -1, :] / temperature
            if top_k > 0:
                v, _ = torch.topk(logits, min(top_k, logits.size(-1)))
                logits[logits < v[:, [-1]]] = float("-inf")
            probs = torch.softmax(logits, dim=-1)
            nxt = torch.multinomial(probs, num_samples=1)
            input_ids = torch.cat([input_ids, nxt], dim=1)
        return input_ids

Encoder – Decoder Models (T5, BART)

The original Transformer used both: an encoder reads the source, a decoder generates the target while cross-attending to the encoder output.Cross-attention is self-attention with queries from the decoder and keys/values from the encoder.

class CrossAttention(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super().__init__()
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads
        self.q_proj = nn.Linear(embed_dim, embed_dim)
        self.k_proj = nn.Linear(embed_dim, embed_dim)
        self.v_proj = nn.Linear(embed_dim, embed_dim)
        self.out_proj = nn.Linear(embed_dim, embed_dim)

    def forward(self, dec_x, enc_out):
        B, Td, C = dec_x.shape; Ts = enc_out.size(1)
        q = self.q_proj(dec_x).view(B, Td, self.num_heads, self.head_dim).transpose(1, 2)
        k = self.k_proj(enc_out).view(B, Ts, self.num_heads, self.head_dim).transpose(1, 2)
        v = self.v_proj(enc_out).view(B, Ts, self.num_heads, self.head_dim).transpose(1, 2)
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.head_dim)
        attn = torch.softmax(scores, dim=-1)
        out = torch.matmul(attn, v).transpose(1, 2).contiguous().view(B, Td, C)
        return self.out_proj(out)

Training a Small Italian LLM

We now take the decoder-only architecture and train it on Italian data. We pull verified datasets from Hugging Face rather than building a corpus from scratch.Chat: FreedomIntelligence/alpaca-gpt4-italian, evol-instruct-italian, mii-llm/ultrafeedback-ita.
Reasoning: raicrits/Orca_ITA_200k, DeepMount00/CulturaViva-ITA.

Tokenizer

We train a BPE tokenizer from scratch for memory efficiency on Italian text.A small 8k vocab keeps the embedding matrix compact — critical for running without a GPU.

from tokenizers import Tokenizer
from tokenizers.models import BPE
from tokenizers.trainers import BpeTrainer
from tokenizers.pre_tokenizers import Whitespace
from tokenizers.decoders import BPEDecoder

def train_tokenizer(files, vocab_size=8192, save_path="italian_tokenizer.json"):
    tok = Tokenizer(BPE(unk_token="<unk>"))
    tok.pre_tokenizer = Whitespace()
    tok.decoder = BPEDecoder()
    trainer = BpeTrainer(vocab_size=vocab_size,
        special_tokens=["<pad>", "<bos>", "<eos>", "<unk>"], min_frequency=2)
    tok.train(files, trainer)
    tok.save(save_path)
    return tok

Data Pipeline

import json, torch
from torch.utils.data import Dataset, DataLoader

class InstructionDataset(Dataset):
    def __init__(self, jsonl_path, tokenizer, max_len=512):
        self.data = [json.loads(l) for l in open(jsonl_path)]
        self.tok = tokenizer; self.max_len = max_len
        self.bos = tokenizer.token_to_id("<bos>")
        self.eos = tokenizer.token_to_id("<eos>")
        self.pad = tokenizer.token_to_id("<pad>")

    def __len__(self): return len(self.data)

    def __getitem__(self, idx):
        item = self.data[idx]
        text = f"Instruction: {item['instruction']}\nResponse: {item['output']}"
        ids = [self.bos] + self.tok.encode(text).ids + [self.eos]
        ids = ids[:self.max_len]
        padded = ids + [self.pad] * (self.max_len - len(ids))
        labels = list(ids) + [-100] * (self.max_len - len(ids))
        return {"input_ids": torch.tensor(padded), "labels": torch.tensor(labels)}

Model Configuration & Training

from dataclasses import dataclass
import torch.nn as nn
from torch.optim import AdamW
from torch.optim.lr_scheduler import CosineAnnealingLR

@dataclass
class DeltaLLMConfig:
    vocab_size: int = 8192
    embed_dim: int = 256
    num_heads: int = 4
    num_layers: int = 4
    ff_dim: int = 1024
    max_len: int = 512

def train(model, dataloader, epochs=3, lr=3e-4, device="cuda" if torch.cuda.is_available() else "cpu"):
    model.to(device).train()
    opt = AdamW(model.parameters(), lr=lr, weight_decay=0.01)
    sched = CosineAnnealingLR(opt, T_max=len(dataloader) * epochs)
    crit = nn.CrossEntropyLoss(ignore_index=-100)
    for epoch in range(epochs):
        for batch in dataloader:
            ids = batch["input_ids"].to(device); labels = batch["labels"].to(device)
            logits = model(ids)[:, :-1, :].contiguous()
            labels = labels[:, 1:].contiguous()
            loss = crit(logits.view(-1, logits.size(-1)), labels.view(-1))
            opt.zero_grad(); loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            opt.step(); sched.step()
    return model

cfg = DeltaLLMConfig()
model = MiniDecoder(cfg.vocab_size, cfg.embed_dim, cfg.num_heads, cfg.num_layers, cfg.ff_dim, cfg.max_len)
tokenizer = train_tokenizer(["italian_corpus.txt"], vocab_size=8192)
ds = InstructionDataset("italian_instructions.jsonl", tokenizer, max_len=512)
dl = DataLoader(ds, batch_size=16, shuffle=True)
model = train(model, dl, epochs=3)
torch.save(model.state_dict(), "deltallm.pt")

Production Path: Unsloth on Colab

For a production fine-tune on a Colab T4, Unsloth cuts VRAM by up to 80% and exports directly to .gguf for Ollama.Colab dumps a fresh itallm-5.2-q4_K_M.gguf straight into your workspace.


from unsloth import FastLanguageModel
from trl import SFTTrainer
from transformers import TrainingArguments

model, tokenizer = FastLanguageModel.from_pretrained(
    model_name="unsloth/Qwen2.5-7B-Instruct", max_seq_length=2048, load_in_4bit=True)

model = FastLanguageModel.get_peft_model(model, r=16,
    target_modules=["q_proj","k_proj","v_proj","o_proj","gate_proj","up_proj","down_proj"],
    lora_alpha=16, lora_dropout=0, bias="none")

trainer = SFTTrainer(model=model, tokenizer=tokenizer, train_dataset=italian_dataset,
    args=TrainingArguments(output_dir="outputs", per_device_train_batch_size=2,
        gradient_accumulation_steps=4, num_train_epochs=3, learning_rate=2e-4,
        fp16=True, logging_steps=10))

trainer.train()

model.save_pretrained_gguf("itallm-5.2", tokenizer, quantization_method="q4_k_m")
